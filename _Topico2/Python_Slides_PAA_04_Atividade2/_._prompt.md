double foo (int* v, int n, int p ) {
      if( n<=0 )
            return 0;
      int i, soma = 0;
      for( int i=0; i<n; i=i+p )
            soma += v[i];
      return sqrt (soma) + foo ( v, soma%n, p ) ;
}
